 37     $W.GLSL.util.handleIncludes = function(source) {
 38         var maxIncludeDepth = 32;
 39         var includeDepth = 0;
 40 
 41 
 42         var includes = source.match(/^#pragma\s*INCLUDE.*/gm);
 43         if (includes !== null) {
 44         }
 45         while (includes !== null && includeDepth < maxIncludeDepth) {
 46             for (var i = 0; i < includes.length; i++) {
 47                 var include = includes[i].replace(/^#pragma\s*INCLUDE\s*/g, "");
 48                 include = include.replace(/["'<>]/g,"");

 57         }

189         var maxLengthSquared = maxLength * maxLength;
190         var vecLengthSquared = V3.lengthSquared(vec);
191 
192         if (vecLengthSquared <= maxLengthSquared) {
193             return vec;
194         }else {
195             return V3.scale(vec, (maxLength / Math.sqrt(vecLengthSquared)));
196         }

257     $W.OS.vecLimitDeviationAngleUtility= function(isInside, source, cosineOfConeAngle, basis) {
258         // immediately return zero length input vectors
259         var sourceLength = V3.length(source);
260         if (sourceLength == 0) {
261             return source;
262         }
263 
264         // measure the angular diviation of "source" from "basis"
265         var direction = V3.scale(source, 1 / sourceLength);
266         var cosineOfSourceAngle = V3.dot(direction, basis);
267 
268         // Simply return "source" if it already meets the angle criteria.
269         if (isInside) {
270             // source vector is already inside the cone, just return it
271             if (cosineOfSourceAngle >= cosineOfConeAngle) return source;
272 
273         }else {
274             // source vector is already outside the cone, just return it
275             if (cosineOfSourceAngle <= cosineOfConeAngle) return source;
276         }
277 
278         // find the portion of "source" that is perpendicular to "basis"
279         var perp = V3.perpendicularComponent(source, basis);
280 
281         // normalize that perpendicular
282         var unitPerp = V3.normalize(perp);
283 
284         // construct a new vector whose length equals the source vector,
285         // and lies on the intersection of a plane (formed the source and
286         // basis vectors) and a cone (whose axis is "basis" and whose
287         // angle corresponds to cosineOfConeAngle)
288         var perpDist = Math.sqrt(1 - (cosineOfConeAngle * cosineOfConeAngle));
289         var c0 = V3.scale(basis, cosineOfConeAngle);
290         var c1 = V3.scale(unitPerp, perpDist);
291 
292         return V3.scale( (V3.add(c0, c1)), sourceLength);
293     };

448         this.regenerateLocalSpaceForBanking = function(newVelocity, elapsedTime) {
449 
450             // the length of this global-upward-pointing vector controls the vehicle's
451             // tendency to right itself as it is rolled over from turning acceleration
452             var globalUp = V3.$(0, 0.5, 0);
453 
454             // acceleration points toward the center of local path curvature, the
455             // length determines how much the vehicle will roll while turning
456             var accelUp = V3.scale(smoothedAcceleration, 0.00);
457 
458             // combined banking, sum of UP due to turning and global UP
459             var bankUp = V3.add(accelUp, globalUp);
460 
461             // blend bankUp into vehicle's UP basis vector
462             var smoothRate = elapsedTime * 3;
463             var tempUp = this.up();
464 
465             tempUp = $W.util.blendVector(smoothRate, bankUp, tempUp);
466             
467             this.setUp(V3.normalize(tempUp));
468 
469             // adjust orthonormal basis vectors to be aligned with new velocity
470             if (this.speed() > 0) {
471                 this.regenerateOrthonormalBasisUF(
472                         V3.scale(newVelocity, 1 / this.speed()));
473             }
474         }

476         this.applySteeringForce = function(force, elapsedTime) {
477             var adjustedForce = this.adjustRawSteeringForce(force, elapsedTime);
478 
479             // enforce limit on magnitude of steering force
480             var clippedForce = V3.truncateLength(adjustedForce, this.maxForce());
481 
482             // compute acceleration and velocity
483             var newAcceleration = V3.scale(clippedForce, 1 / this.mass());
484             var newVelocity = this.velocity();
485 
486             // damp out abrupt changes and oscillations in steering acceleration
487             // (rate is proportional to time step, then clipped into useful range)
488             if (elapsedTime > 0) {
489                 var smoothRate = $W.util.clip(9 * elapsedTime, 0.15, 0.4)
490                 smoothedAcceleration = 
491                     $W.util.blendVector(smoothRate, newAcceleration, smoothedAcceleration);
492                 
493             }
494             // Euler integrate (per frame) acceleration into velocity
495             V3.add(V3.scale(smoothedAcceleration, elapsedTime), newVelocity, newVelocity);
496 
497             // enforce speed limit
498             newVelocity = V3.truncateLength(newVelocity, this.maxSpeed());
499 
500             // update Speed
501             this.setSpeed (V3.length(newVelocity));
502 
503 
504             // Euler integrate (per frame) velocity into position
505             this.setPosition(V3.add(this.position(), 
506                                     V3.scale(newVelocity, elapsedTime)));
507 
508 
509             // regenerate local space (by default: align vehicle's forward axis with
510             // new velocity, but this behavior may be overridden by derived classes.)
511             this.regenerateLocalSpace (newVelocity, elapsedTime);
512 
513             // maintain path curvature information
514             // XXX implement
515             //measurePathCurvature (elapsedTime);
516 
517             // running average of recent positions
518             smoothedPosition = $W.util.blendVector(elapsedTime * 0.06,
519                                                    this.position(),
520                                                    this.smoothedPosition());
521             smoothedPosition = this.position();
522 
523         }
524 
525         this.reset();
526     };

596     $W.OS.steering=function steering() {
597         this.steerToFlock = function OS_steerToFlock(flock) {
598             var radiusScale = 2;
599             var separationRadius =  5.0 * radiusScale;
600             var separationAngle  = -0.707;
601             var separationWeight =  8.0;
602             
603             var alignmentRadius = 7.5 * radiusScale;
604             var alignmentAngle  = 0.7;
605             var alignmentWeight = 10.0;
606              
607             var cohesionRadius = 9.0 * radiusScale;
608             var cohesionAngle  = -0.15;
609             var cohesionWeight = 10.0;
610 
611             var neighbors = [];
612 
613             for (var i = 0; i < flock.length; i++) {
614                 var other = flock[i];
615 
616                 if (this.inBoidNeighborhood(other, separationRadius, cohesionRadius, 
617                                              alignmentAngle)) {
618                     neighbors.push(other);
619                 }
620             }
621 
622             var separation = this.steerForSeparation (separationRadius, separationAngle, neighbors);
623             var alignment  = this.steerForAlignment  (alignmentRadius, alignmentAngle, neighbors);
624             var cohesion   = this.steerForCohesion   (cohesionRadius, cohesionAngle, neighbors);
625 
626             separation = V3.scale(separation, separationWeight);
627             alignment = V3.scale(alignment, alignmentWeight);
628             cohesion = V3.scale(cohesion, cohesionWeight);
629 
630             return  V3.add(separation, V3.add(alignment, cohesion));
631         }
632 
633         this.steerForSeparation = function OS_steerForSeparation(maxDistance, cosMaxAngle, flock) {
634             // steering accumulator and count of neighbors, both initially zero
635             var steering = V3.$(0,0,0);
636             var neighbors = 0;
637 
638             // for each of the other vehicles...
639             for (var i = 0; i < flock.length; i++) {
640                 var other = flock[i];
641 
642                 if (this.inBoidNeighborhood(other, this.radius() * 3, maxDistance, cosMaxAngle)) {
643                     
644                     // add in steering contribution
645                     // (opposite of the offset direction, divided once by distance
646                     // to normalize, divided another time to get 1/d falloff)
647                     var offset = V3.sub(other.position(), this.position());
648                     var invNegDistSquared = 1 / -(V3.dot(offset, offset));
649                     var scaledOffset = V3.scale(offset, invNegDistSquared);
650                     steering = V3.add(scaledOffset, steering);
651 
652                     // count neighbors
653                     neighbors++;
654                 }
655             }
656 
657             steering[1] = 0.1 * steering[1];
658             steering = V3.normalize(steering);
659             return steering;
660         }

662         this.steerForAlignment = function OS_steerForAlignment(maxDistance, cosMaxAngle, flock) {    
663             var steering = V3.$(0,0,0);
664             var neighbors = 0;
665 
666             // for each of the other vehicles...
667             for (var i = 0; i < flock.length; i++) {
668                 var other = flock[i];
669 
670                 if (this.inBoidNeighborhood(other, this.radius() * 3, maxDistance, cosMaxAngle)) {
671 
672                     // accumulate sum of neighbor's heading
673                     V3.add(other.forward(), steering, steering);
674 
675                     // count neighbors
676                     neighbors++;
677                 }
678             }
679 
680             // divide by neighbors, subtract off current position to get error-
681             // correcting direction, then normalize to pure direction
682             if (neighbors > 0) {
683                 steering = V3.normalize( 
684                                     V3.sub( V3.scale(steering, 1 / neighbors),       
685                                             this.forward()));
686             }
687 
688             return steering;
689         }

696             for (var i = 0; i < flock.length; i++) {

705                 }

711             if (neighbors > 0) {
712                 steering = V3.normalize( 
713                                     V3.sub( V3.scale(steering, 1 / neighbors),       
714                                             this.position()));

722         this.inBoidNeighborhood = function OS_inBoidNeighborhood(other, minDistance, maxDistance, cosMaxAngle) {
723             if (other === this) {
724                 return false;
725 
726             }else {
727                 var offset = V3.sub(other.position(), this.position());
728                 var distanceSquared = V3.lengthSquared(offset);
729 
730                 // definitely in neighborhood if inside minDistance sphere
731                 if (distanceSquared < (minDistance * minDistance)){
732                     return true;
733                 }else {
734                     // definitely not in neighborhood if outside maxDistance sphere
735                     if (distanceSquared > (maxDistance * maxDistance)) {
736                         return false;
737                     }else {
738                         // otherwise, test angular offset from forward axis
739                         var unitOffset = V3.scale(offset, 1 / Math.sqrt(distanceSquared));
740                         var forwardness= V3.dot(this.forward(), unitOffset);
741                         return forwardness > cosMaxAngle;
742                     }
743                 }
744             }
745         }

764     $W.SimulatedObject = function(physType) {
765         $W.Object.call(this, $W.GL.TRIANGLES);
766 
767         if (typeof($W.world) == 'undefined') {
768             $W.world = { 
769                 objects:[],
770                 age:0,

773                     // Cleared per frame
774                     for (var i = 0; i < this.objects.length; i++) {
775                         this.objects[i].alreadyCollidedWith = [];
776                     }
777                     for (var i = 0; i < this.objects.length; i++) {
778                         this.objects[i].updateSim(dt);
779                     }
780                 }
781             }
782         }
783         this.index = $W.world.objects.length;
784         $W.world.objects.push(this); // Add this to global physical object list
785 
786         /** Forces to be applied this frame.
787          * Cleared with each call to update()
788          */
789         this.impulses = [];
790 
791         this.alreadyCollidedWith = [];
792 
793         /** Type of object to simulate
794          * "sphere" or "wall" are valid
795          */
796         this.physType = physType;
797 
798         /** Bounding sphere radius */
799         this.radius = 5;
800         /** Object mass */
801         this.mass = 1;
802         /** Object velocity */
803         this.velocity = Vector.Zero(3);
804         /** Angular velocity */
805         this.omega = Vector.Zero(3);
806         /** Coefficient of restitution */
807         this.restitution = 1;
808 
809         this.maxSpeed = 0.02;

811         this.applyImpulse = function(impulse) {
812             this.impulses.push(impulse);
813         }

815         this.shouldTestCollisionWith = function(object) {
816             // Walls don't collide
817             if (this.physType == "wall" && object.physType == "wall"){
818                 return false;
819             }
820 
821             // Don't test collision against self
822             if (this.index === object.index) {
823                 return false;
824             }
825 
826             // Don't recalc collisions if we've already tested
827             for (var i = 0; i < object.alreadyCollidedWith.length; i++) {
828                 if (this.index === object.alreadyCollidedWith[i]) {
829                     return false;
830                 }
831             }
832 
833             return true;
834         }

836         this.tempSphereForWallCollision = function(other) {
837             var sphere = {};
838             sphere.velocity = this.velocity;
839             sphere.impulses = [];
840             sphere.applyImpulse = function(){};
841             sphere.radius = this.radius;
842             sphere.position = this.line.pointClosestTo(other.position);
843             sphere.physType = "wall";
844             return sphere;
845         }

847         this.performCollisions = function(dt) {
848             for (var i = 0; i < $W.world.objects.length; i++) {
849                 var object = $W.world.objects[i];
850 
851                 if (this.shouldTestCollisionWith(object) && this.isColliding(object)) {
852                     var a = this;
853                     var b = object;
854 
855                     if (a.physType == "wall") {
856                         a = a.tempSphereForWallCollision(b);
857                     }
858                     if (b.physType == "wall") {
859                         b = b.tempSphereForWallCollision(a);
860                     }
861 
862                     var t = collisionDeltaOffset(this, object);
863                     dt += t;
864 
865                     var actionNormal = a.position.subtract(b.position).toUnitVector();
866                     a.position = a.position.add(b.velocity.x(t));
867                     b.position = b.position.add(a.velocity.x(t));
868 
869                     var vDiff = a.velocity.subtract(b.velocity);
870                     var relativeNormalVelocity = vDiff.dot(actionNormal);
871 
872                     var impulse = actionNormal.x(relativeNormalVelocity);
873                     object.applyImpulse(impulse.x(1 / object.mass).x(this.restitution));
874                     this.applyImpulse(impulse.invert().x(1 / this.mass).x(object.restitution));
875                 }
876 
877                 object.alreadyCollidedWith.push(this.index);
878             }
879         }
880 
881         this.updateSim = function(dt) {
882             //--- calculate forces ---
883             this.performCollisions(dt);
884 
885 
886             //--- calculate velocities ---
887             // Apply accrued impulses
888             while (this.impulses.length > 0) {
889                 this.velocity = this.velocity.add(this.impulses.pop());
890             }
891 
892             var temp = V3.$(this.velocity.e(1), this.velocity.e(2), this.velocity.e(3));
893             //this.velocity.elements = V3.elements(V3.truncateLength(temp, this.maxSpeed));
894 
895             //--- integrate position/rotation ---
896             this.position = this.position.add(this.velocity.x(dt));
897         }

899         var areColliding = function(a, b) {
900             return didCollide(a, b, 0);
901         }

903         var collisionDeltaOffset = function(a, b) {
904             // Back up to when the collision occurred
905             // Otherwise objects could get stuck inside one another
906             var t = 0;
907             while (t < $W.timer.dt && didCollide(a, b, t++)) {}
908             return t; // to catch back up
909         }
910 
911         var didCollide = function(a, b, t) {
912             if (a.physType == "wall" && b.physType == "wall") { return false; }
913             var pa;
914             var pb;
915 
916             if (a.physType == "sphere") {
917                 pa = a.position;
918             }else if (a.physType == "wall") {
919                 pa = a.line.pointClosestTo(b.position);
920             }
921 
922             if (b.physType == "sphere") {
923                 pb = b.position;
924             }else if (b.physType == "wall") {
925                 pb = b.line.pointClosestTo(a.position);
926             }
927 
928             return $W.util.sphereCollide(
929                     pa.subtract(a.velocity.x(t)),
930                     pb.subtract(b.velocity.x(t)), 
931                     a.radius, b.radius);
932         }
