  3     /** @class Common state representation (position, rotation, scale).
  4      * A position, rotation, and scale.
  5      * @param {3 Array} pos Position.
  6      * @param {3 Array} rot Rotation.
  7      * @param {3 Array} scl Scale.
  8      */
  9     $W.ObjectState = function(position, rotation, scale) {
 10         if (arguments.length == 3) {
 11             this.position  = $V(position);
 12             this.rotation  = $V(rotation);
 13             this.scale      = $V(scale);
 14         }else {
 15             this.position  = Vector.Zero(3);
 16             this.rotation  = Vector.Zero(3);
 17             this.scale      = $V([1,1,1]);
 18         }
 19 
 20         this.recalculateQuaternion = function() {
 21             // No need to calc for no rotation
 22             if (this.rotation.eql(Vector.Zero(3))) {
 23                 this.q = new $W.Quaternion();
 24 
 25             }else {
 26                 var qh = new $W.Quaternion(0, 1, 0, this.rotation.e(1));
 27                 var qp = new $W.Quaternion(1, 0, 0, this.rotation.e(2));
 28                 var qr = new $W.Quaternion(0, 0, 1, this.rotation.e(3));
 29 
 30                 this.q = (qr.multiply(qp)).multiply(qh);
 31             }
 32         }
 33 
 34         this.recalculateQuaternion();
 35 
 36         /** Set a new position. */
 37         this.setPosition = function(x, y, z) { 
 38             // Called as setPosition(x, y, z)
 39             if (arguments.length == 3) {
 40                 this.position.elements = [x, y, z]; 
 41 
 42                 // Called as setPosition([x, y, z])
 43             }else {
 44                 this.position.elements = arguments[0];
 45             }
 46         }
 47 
 48         /** Set a new rotation. */
 49         this.setRotation = function() { 
 50             // Called as setRotation(x, y, z)
 51             if (arguments.length == 3) {
 52                 this.rotation.elements = arguments; 
 53 
 54                 // Called as setRotation([x, y, z])
 55             }else {
 56                 this.rotation.elements = arguments[0];
 57             }
 58 
 59             this.recalculateQuaternion();
 60         }
 61 
 62         /** Set a new scale. */
 63         this.setScale = function(x, y, z){ 
 64             this.scale.elements    = [x, y, z]; 
 65         }
 66 
 67         /** Set the x y and z components of the object's scale to the given
 68          * value.
 69          * @param {Number} s New scale of the object.
 70          */
 71         this.setScaleUniformly = function(s) { 
 72             this.scale = $V([s,s,s]); 
 73         };

 77             if ((other.scale !== undefined &&
 78                         other.position !== undefined &&

 82                             this.position == other.position &&

 85                 return true;
 86             }else {
 87                 return false;

 92     /** @class Storage and handling of GL data arrays
 93      * @param {String} name Attribute name data is for
 94      */
 95     $W.ArrayBuffer = function(name, data) {
 96         this.name = name;
 97         this.data = data;
 98         this.glData = new WebGLFloatArray(data);
 99         this.glBuffer = $W.GL.createBuffer();
100 
101         this.setData = function ABUF_setData(data) {
102             this.data = data;
103             this.glData = new WebGLFloatArray(data);
104             this.buffer();
105         };
106 
107         this.buffer = function ABUF_buffer() {
108             try {
109                 this.bind();
110                 $W.GL.bufferData($W.GL.ARRAY_BUFFER, this.glData, $W.GL.STATIC_DRAW);
111                 this.unbind();
112             }catch (e) {
113                 console.error(e);
114             }
115         };
116 
117         this.associate = function ABUF_associate(attrib) {
118             try {
119                 this.bind();
120                 $W.GL.enableVertexAttribArray(attrib.location);
121                 $W.GL.vertexAttribPointer(attrib.location, attrib.length,
122                         $W.GL.FLOAT, false, 0, 0);
123                 this.unbind();
124             }catch (e) {
125                 console.error("Failed to associate array buffer `" + this.name + 
126                         "` with vertex attribute `" + attrib.name + "`");
127                 console.dir(this, attrib);
128                 console.error(e);
129             }
130         };
131 
132         this.bind = function ABUF_bind() {
133             $W.GL.bindBuffer($W.GL.ARRAY_BUFFER, this.glBuffer);
134         };
135         this.unbind = function ABUF_unbind() {
136             $W.GL.bindBuffer($W.GL.ARRAY_BUFFER, null);
137         };

141     $W.RENDERABLE   = 1;
142     $W.PICKABLE     = 2;
143     $W.createObject = function(params) {
144         var obj = new $W.Object(params.type);
145         var has = function(param){
146             return typeof(param) !== 'undefined';
147         };
148 
149         if (has(params.material)){
150             obj.setMaterial(params.material);
151         }
152 
153         if (has(params.model)){
154             obj.fromModel(params.model);
155         }else if (has(params.data)){
156             obj.fillArrays(params.data);
157         }
158 
159         if (has(params.vertexCount)){
160             obj.vertexCount = params.vertexCount;
161         }
162 
163         return obj;
164     };

167     /** @class Contains pertinent render information for an individual renderable entity.
168      *
169      * Make sure to set vertexCount correctly
170      * Animations are clunky right now, I'm working on it.
171      *
172      * @param type The type of rendering to use for this object, possible values
173      * are:<br>
174      * $W.GL.POINTS         <br>
175      * $W.GL.LINES          <br>
176      * $W.GL.LINE_LOOP      <br>
177      * $W.GL.LINE_STRIP     <br>
178      * $W.GL.TRIANGLES      <br>
179      * $W.GL.TRIANGLE_STRIP <br>
180      * $W.GL.TRIANGLE_FAN   <br>
181      * @param {Boolean} shouldAdd Set to false to not add this the the object
182      * list in case you want to handle rendering in a specific manner, e.g. as
183      * the child of another object.
184      */
185     $W.Object = function (type, flags) {
186         //console.group("Creating object");
187         $W.ObjectState.call(this);
188 
189         $W.objects.push(this);
190 
191         if (typeof(flags) === 'undefined' ||/*backcompat*/ flags === true) {
192             flags = $W.RENDERABLE | $W.PICKABLE;
193         }/*backcompat*/else if(flags === false) {
194             flags = $W.PICKABLE;
195         }
196 
197 
198         if (flags & $W.RENDERABLE){
199             $W.renderables.push(this);
200         }
201 
202         if (flags & $W.PICKABLE) {
203             $W.pickables.push(this);
204         }
205 
206 
207         /** Number of vertices in this object.
208          * Used when rendering with drawArrays.
209          */
210         this.vertexCount = 0;
211 
212         this.id = $W.createdObjectCount++;
213 
214         /* The type of rendering to use for this object */
215         this.type = type; 
216 
217         this.material = $W.materials['wglu_default'];
218 
219         this.arrayBuffers = [];
220 
221         this.children = [];
222 
223         /** The animation for this object. */
224         this.animation = new $W.ProceduralAnimation();
225 
226         // drawArrays by default
227         this._drawFunction = $W.renderer.drawArrays;
228 
229         /** Name of shader program used to render this object */
230         /** Add an object as a child to this object.
231          * @param {Object} obj The object to add as a child.
232          */
233         this.addChild = function(obj) {
234             this.children.push(obj);
235         };
236 
237         /** Set the indices for the elements of this object.
238          * Draws this object with drawElements unless set with
239          * false.
240          * @param {Array|Boolean} elements The array of indices of the
241          * elements or false, which disabled drawElements rendering
242          * for this object.
243          */
244         this.setElements = function(elements) {
245             // don't use drawElements
246             if (elements === false) {
247                 this.arrayBuffers.wglu_internal_elements = undefined;
248                 $W.GL.bindBuffer($W.GL.ELEMENT_ARRAY_BUFFER, null);
249 
250             // use drawElements
251             }else {
252                 this._drawFunction = $W.renderer.drawElements;
253 
254                 var elementAB = new $W.ArrayBuffer('wglu_internal_elements',
255                         $W.util.flattenArray(elements));
256 
257                 $W.GL.bindBuffer($W.GL.ELEMENT_ARRAY_BUFFER, elementAB.glBuffer);
258                 $W.GL.bufferData($W.GL.ELEMENT_ARRAY_BUFFER,
259                         new WebGLUnsignedShortArray( elementAB.data), 
260                         $W.GL.STATIC_DRAW);
261 
262                 this.arrayBuffers.wglu_internal_elements = elementAB;
263             }
264         };
265 
266         /** Set the material of this object to the indicated material
267          *
268          * @param material {String|Material} Either the name of an existing material
269          * or a Material object.
270          */
271         this.setMaterial = function OBJ_setMaterial(material) {
272             if (typeof(material) === 'string') { 
273                 this.material = $W.materials[material];
274             }else {
275                 this.material = material;
276             }
277         };

279         /** Fills the array of the given name, where name is a 
280          * vertex attribute in the shader. 
281          * Also creates a buffer to hold the data in WebGL.
282          * @param {String} name The attribute variable name in a shader
283          * attached to the shader program used by this object. (this is
284          * not verified for you)
285          * @param {Array} contents The data to pass to the attribute.
286          */
287         this.fillArray = function OBJ_fillArray(name, data) {
288             data = $W.util.flattenArray(data);
289             if (typeof(this.arrayBuffers[name]) === 'undefined') {
290                 this.arrayBuffers[name] = new $W.ArrayBuffer(name, data);
291             }else {
292                 this.arrayBuffers[name].setData(data);
293             }
294 
295             this.arrayBuffers[name].buffer();
296         };
297 
298         this.fillArrays = function OBJ_fillArrays(arrays) {
299             for (var i = 0; i < arrays.length; i++) {
300                 var arr = arrays[i];
301                 if (arr[0] === 'wglu_elements') {
302                     this.setElements(arr[1]);
303                 }else {
304                     this.fillArray(arr[0], arr[1]);
305                 }
306             }
307         };
308 
309         this.fromModel = function OBJ_fromModel(model) {
310             this.fillArrays([['vertex', model.vertices],
311                              ['normal', model.normals],
312                              ['texCoord', model.texCoords],
313                              ['wglu_elements', model.indices]]);
314 
315         };

317         /** draw this object at the given postion, rotation, and scale
318          * @param {3 Element Array} pos Position array.
319          * @param {Matrix} rot Rotation matrix.
320          * @param {3 Element Array} scale Scaling array.
321          */
322         this.drawAt = function OBJ_drawAt(pos, rot, scale) {
323                 $W.modelview.push();
324 
325                 $W.modelview.translate(pos);
326                 $W.modelview.multiply(rot);
327                 $W.modelview.scale(scale);
328 
329                 for (var i = 0; i < this.children.length; i++) {
330                     this.children[i].draw();
331                 }
332 
333                 $W.renderer.renderObject(this, this.material, this._drawFunction);
334 
335                 $W.modelview.pop();
336                 $W.GL.bindTexture($W.GL.TEXTURE_2D, null);
337         };
338 
339         this.drawChildrenAt = function(pos, rot, scale) {
340                 $W.modelview.pushMatrix();
341 
342                 $W.modelview.translate(pos);
343                 $W.modelview.multMatrix(rot);
344                 $W.modelview.scale(scale);
345 
346                 for (var i = 0; i < this.children.length; i++) {
347                     this.children[i].draw();
348                 }
349 
350                 $W.modelview.popMatrix();
351         };
352 
353         /** draw this object at its internally stored position, rotation, and
354          * scale, INCLUDING its current animation state.
355          */
356         this.draw = function OBJ_draw() {
357             this.drawAt(
358                 this.animatedPosition().elements, 
359                 this.animatedRotation().matrix(),
360                 this.animatedScale().elements
361             );
362         };
363 
364         this.drawChildren = function() {
365             this.drawChildrenAt(
366                 this.animatedPosition().elements, 
367                 this.animatedRotation().matrix(),
368                 this.animatedScale().elements
369             );
370         };
371 
372         /** Update this object's animation state. 
373          * @param {Number} dt The delta time since the previous call to
374          * update.
375          */
376         this.update = function(dt) {
377             this.animation.update(dt);
378 
379             for (var i = 0; i < this.children.length; i++) {
380                 this.children[i].update(dt);
381             }
382         };

385         /** @returns {Vector} The sum of the object's base position and its 
386          * animation.
387          */
388         this.animatedPosition = function() { 
389             return this.position.add(this.animation.position); 
390         };
391 
392         /** @returns {Vector} The sum of the object's base rotation and its 
393          * animation. 
394          */
395         this.animatedRotation = function() { 
396             //return this.rotation.add(this.animation.rotation); 
397             return this.q.multiply(this.animation.q);
398         };
399 
400         /** @returns {Vector} The product of the object's base scale and its 
401          * animation. 
402          */
403         this.animatedScale    = function() { 
404             return $V([
405                 this.scale.e(1) * this.animation.scale.e(1),
406                 this.scale.e(2) * this.animation.scale.e(2),
407                 this.scale.e(3) * this.animation.scale.e(3)
408             ]);
409         };
