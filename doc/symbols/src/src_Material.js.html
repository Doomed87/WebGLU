1 /** @ignore Wrapper function to allow multifile or single file organization */
  2 $W.initMaterial = function() {
  3 
  4     /* @class Creates a new material from the provided definition.
  5      * Definition is JS object (possibly loaded from JSON) in
  6      * the following format
  7      * {
  8      *  name: "matName", // The name this material will be refered to by.
  9      *                   // Must be unique within this application
 10      *
 11      *  program: {
 12      *      name: "prgName", // If there already exists a shader program
 13      *                       // with this name, it will be used. Otherwise
 14      *                       // one will be created as defined here.
 15      *      shaders: [ // List of shaders in this program.
 16      *                 // As with the program itself, if a shader with the
 17      *                 // given name already exists it will take precedence
 18      *                 // over creating a new shader.
 19      *          {name: "FS", path: "FS.frag"},
 20      *          {name: "VS", type: $W.GL.VERTEX_SHADER, source: "..."},
 21      *          // etc.
 22      *      ]
 23      *  },
 24      *
 25      *  textures: [
 26      *      // type can be `Image`, `Video`, or `Canvas`
 27      *      {name:"tex", type:"Image", path:"tex.jpg"},
 28      *      // etc.
 29      *  ]
 30      * }
 31      * or
 32      * {
 33      *  path: "mat.json"
 34      * }
 35      * where mat.json is a file containing JSON in the first format
 36      */
 37     $W.Material = function(materialDef) {
 38 
 39         this.name = null;
 40         this.program = null;
 41         this.uniforms   = [];
 42         this.attributes = [];
 43         this.textures   = [];
 44         this.framebuffer= null;
 45 
 46         this.setUniformAction = function MAT_setUniformAction(name, action) {
 47             var uniform;
 48             for (var i = 0; i < this.uniforms.length; i++) {
 49                 if (this.uniforms[i].name === name) {
 50                     uniform = this.uniforms[i];
 51                 }
 52             }
 53             if (uniform !== null) { 
 54                 uniform.action = action;
 55             }else {
 56                 console.warn("Cannot set uniform action for nonexistant uniform `"+
 57                         name + "`");
 58             }
 59         };
 60 
 61         this.setupUniforms = function MAT_setupUniforms() {
 62             this.uniforms = [];
 63             this.attributes = [];
 64 
 65             for (var i = 0; i < this.program.uniforms.length; i++) {
 66                 this.uniforms.push(this.program.uniforms[i].clone());
 67             }
 68 
 69             for (var i = 0; i < this.program.attributes.length; i++) {
 70                 var attrib = this.program.attributes[i];
 71 
 72                 if (attrib.location !== -1) {
 73                     this.attributes.push(attrib.clone());
 74                     $W.GL.enableVertexAttribArray(attrib.location);
 75 
 76                 }
 77             }
 78 
 79         };
 80 
 81         this.setupTextureUniforms = function MAT_setupTextureUniforms() {
 82             for (var i = 0; i < this.textures.length; i++) {
 83                 var action = genMultiTextureAction(i);
 84                 this.setUniformAction('wglu_mat_texture'+i, action);
 85             }
 86         };
 87 
 88         this.initFromDef = function MAT_initFromDef(matDef) {
 89             // If we've been provided a path, use that file.
 90             // Allow for name override.
 91             if (typeof(matDef.path) !== 'undefined') {
 92                 if (typeof(matDef.name) !== 'undefined') {
 93                     this.name = matDef.name;
 94                     matDef = $W.util.loadFileAsJSON(matDef.path);
 95                 }else {
 96                     matDef = $W.util.loadFileAsJSON(matDef.path);
 97                     this.name = matDef.name;
 98                 }
 99             }else {
100                 this.name = matDef.name;
101             }
102 
103             $W.log(matDef.name);
104 
105             $W.log('creating material `' + this.name + '`');
106             $W.indentLog();
107 
108             var initProgram = function MAT_initProgram(prgDef) {
109                 // If a program of the name specified in the material doesn't already
110                 // exist, create it with the specified parameters.
111                 var program = $W.programs[prgDef.name];
112                 if (typeof(program) === 'undefined') {
113 
114                     program = new $W.GLSL.ShaderProgram(prgDef.name);
115 
116                     for (var i = 0; i < prgDef.shaders.length; i++) {
117                         var shaderDef = prgDef.shaders[i];
118 
119                         // Load shader from file
120                         if (typeof(shaderDef.path) !== 'undefined') {
121                             program.attachShader(shaderDef.name, shaderDef.path);
122 
123                         // Load shader from source code
124                         }else if (typeof(shaderDef.source) !== 'undefined') {
125                             program.attachShader(shaderDef.name, shaderDef.source, shaderDef.type);
126 
127                         // Load shader from embedded script
128                         }else {
129                             program.attachShader(shaderDef.id);
130                         }
131                     }
132                     program.use();
133                 }
134 
135                 return program
136             };
137 
138             var initTextures = function MAT_initTextures(texDef) {
139                 var textures = []
140 
141                 // If textures specified in the material do not exist, create them.
142                 if (typeof(texDef) !== 'undefined') {
143                     for (var i = 0; i < texDef.length; i++) {
144                         var texture = texDef[i];
145 
146                         if (typeof($W.textures[texture.name]) === 'undefined') {
147 
148                             // type is Image, Canvas, or Video. this calls
149                             // ImageTexture, CanvasTexture or VideoTexture.
150                             if (texture.type === "Image") {
151                                 new $W.ImageTexture(texture.name, texture.path);
152 
153                             }else if (texture.type == "Canvas") {
154                                 new $W.CanvasTexture(texDef.name, 
155                                         document.getElementById(texture.element_id));
156 
157                             }else if (texture.type === "Video") {
158                                 
159                                 new $W.VideoTexture(texture.name, texture.path);
160                             }
161 
162                             textures.push(texture.name);
163                         }
164                     }
165                 }
166                 return textures
167             };
168 
169             this.program = initProgram(matDef.program);
170             this.setupUniforms();
171 
172             this.textures = initTextures(matDef.textures);
173             this.setupTextureUniforms();
174 
175             $W.log(this);
176 
177             $W.dedentLog();
178         };
179 
180         this.initFromDef(materialDef)
181 
182         $W.materials[this.name] = this;
183     };
184 };
