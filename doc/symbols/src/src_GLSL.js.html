 57         getShaderSourceById: function(id) {
 58             var source;
 59             var shaderScript = document.getElementById(id);
 60             if (!shaderScript) {
 61                 console.error("No script with id '" + id + "'");
 62                 return null;
 63             }
 64         
 65             source = "";
 66             var k = shaderScript.firstChild;
 67             while (k) {
 68                 if (k.nodeType === 3) {
 69                     source += k.textContent;
 70                 }
 71                 k = k.nextSibling;
 72             }
 73         
 74             return source;
 75         },

 82                 return null;
 83             }
 84         
 85             if (shaderScript.type === "x-shader/x-fragment") {
 86                 type = $W.GL.FRAGMENT_SHADER;
 87             } else if (shaderScript.type === "x-shader/x-vertex") {
 88                 type = $W.GL.VERTEX_SHADER;
 89             } else {
 90                 console.warn('invalid shader type' + shaderScript.type);
 91             }
 92         
 93             return type;
 94         }

133     /** @class Handles compilation, attributes, and uniforms of a GLSL
134      * shader.
135      * @param {String} name All shaders need a unique name
136      * @param {String} [src] The source code, if not included then
137      * name is assumed to be a DOM element Id to a script element 
138      * containing the source code for a shader.
139      * @param [type] The type of shader, valid types are<br>
140      * $W.GL.VERTEX_SHADER<br>
141      * $W.GL.FRAGMENT_SHADER<br>
142      */
143     $W.GLSL.Shader = function(name, src, type) {
144         $W.log("creating shader '" + name + "'");
145         $W.indentLog();
146         $W.shaders[name] = this;
147 
148         /** Name of the shader. */
149         this.name = name;
150         var name = this.name;
151 
152         /** Source code of the shader. */
153         var source = src;
154 
155         /** Shader type $W.GL.VERTEX_SHADER or $W.GL.FRAGMENT_SHADER. */
156         this.type = -1;
157 
158         // If the source wasn't passed in we assume name is an element
159         // ID in the page
160         if (source === undefined) {
161             source = $W.GLSL.util.getShaderSourceById(name);
162             this.type = $W.GLSL.util.getShaderTypeById(name);
163 
164         // Else we just use the provided source and type
165         }else {
166             this.type = type;
167         }
168 
169         /** The WebGL shader object. */
170         var glShader = null;
171 
172         /** Attributes for this shader. */
173         this.attributes = [];
174         /** Uniforms for this shader. */
175         this.uniforms   = [];
176 
177         /** Names of programs which use this shader. */
178         var programs   = [];
179 
180         /** Tracks if this shader need to be (re)compiled. */
181         var isDirty = true;
182 
183         // TODO Have _dirty and _clean update the dirty status of all
184         // programs which use this shader.
185         var dirty = function() {
186             glShader = null;
187             $W.log("Marking shader `" + name + "` as dirty", $W.LL.INFO);
188             for (var i = 0; i < programs.length; i++) {
189                 $W.log('dirtying program `' + programs[i] + "`", $W.LL.INFO);
190                 $W.programs[programs[i]].dirty();                            
191             }
192             isDirty = true;
193         };
194 
195         var clean = function() {
196             isDirty = false;
197         };
198         this.isDirty = function() {
199             return isDirty;
200         };
201 
202 
203         this.addProgram = function(name) {
204             programs.push(name);
205         };
206         this.removeProgram = function(name) {
207             programs = programs.remove(name);
208         };
209         
210         /** Change the source for this shader 
211          * @param {String} src The source code.
212          */
213         this.setSource = function(src) {
214             dirty();
215             source = src;
216         };
217 
218         this.source = function() {
219             return source;
220         };

222         /** @returns The raw WebGL shader object */
223         this.getGLShader = function() {
224             // Ensure the shader is valid before we return it
225             if (isDirty) {
226                 $W.info("'" + this.name + "' is dirty");
227                 if (!this.compile()) {
228                     return false;
229                 }else {
230                     clean();
231                 }
232             }else {
233                 $W.info("'" + this.name + "' is clean, using");
234             }
235             return glShader;
236         }

238         /** Store the information about this named uniform. 
239          * @param {String} name The uniform variable name as defined in
240          * the shader source.
241          * @param {Function} [action=function(){}] The function to call 
242          * each frame to prep/send this uniform to the shader.
243          */
244         this.addUniform = function(name, action, type) {
245             $W.debug("\tadding uniform '" + name + "'");
246             if (!action) {
247                 action = function(){}
248             }
249             this.uniforms.push(new $W.GLSL.Uniform(name, action, type));
250         }

252         /** Store the information about this named attribute. 
253          * @param {String} name The attribute variable name as defined in
254          * the shader source.
255          * @param {Integer} [length=3] The length of the attribute.
256          */
257         this.addAttribute = function(name, length) {
258             $W.debug("\tadding attribute '" + name + "'");
259             if (!length) { length = 3; }
260             this.attributes.push(new $W.GLSL.Attribute(name, length));
261         }

263         // Find and initialize all uniforms and attributes found in the source
264         this.parseShaderVariables = function(str) {
265             var tokens = str.split(/[\[\]\s\n;]+?/);
266 
267             for (var i = 0; i < tokens.length; i++) {
268                 if (tokens[i] === "attribute") {
269                     var type = tokens[i+1];
270                     var name = tokens[i+2];
271                     var length = $W.GLSL.shaderVarLengths[type];
272                     this.addAttribute(name, length);
273                 }                               
274                 if (tokens[i] === "uniform") {
275                     var type = tokens[i+1];
276                     var name = tokens[i+2];
277 
278                     this.addUniform(name, $W.util.getUniformAction(name), type);
279                 }
280             }
281         };

283         /** Compile the shader if able.
284          * Lets any shader programs which use this shader know they need to
285          * relink
286          */
287         this.compile = function() {
288             if (this.type === $W.GL.VERTEX_SHADER) {
289                 $W.debug("compiling '" + this.name + "' as vertex shader");
290             }else if (this.type === $W.GL.FRAGMENT_SHADER) {
291                 $W.debug("compiling '" + this.name + "' as vertex shader");
292             }else {
293                 console.error("compiling '" + this.name + "' as unknown type " + this.type);
294             }   
295             if (glShader !== null) {
296                 $W.GL.deleteShader(glShader);
297                 glShader = null;
298             }
299 
300             var shader = $W.GL.createShader(this.type);
301             
302             $W.GL.shaderSource(shader, source);
303             $W.GL.compileShader(shader);
304 
305             if (!$W.GL.getShaderParameter(shader, $W.GL.COMPILE_STATUS)) {
306                 console.error("Compile error in `" + this.name + "`\n" +
307                         $W.GL.getShaderInfoLog(shader));
308                 glShader = null;
309             } else {
310                 clean();
311                 glShader = shader;
312             }
313 
314             return (glShader !== null);
315         };
316 
317         //source = $W.GLSL.util.handlePragmas(source);
318         this.parseShaderVariables(source);
319         this.compile();
320         $W.dedentLog();
321     };

323     /** @class Handles data and linking for a shader program.
324      * Also ensures all shaders which it uses are compiled and up
325      * to date.
326      * @param {String} name All shader programs need a unique name.
327      */
328     $W.GLSL.ShaderProgram = function(name) {
329         $W.log("creating shader program '" + name + "'");
330         $W.indentLog();
331 
332         /** Global name of the shader program. */
333         this.name = name;
334         $W.programs[name] = this;
335 
336         /** WebGL shader program object. */
337         this.glProgram  = null;
338 
339         /** Tracks link state and compile state of all attached shaders. */
340         var isDirty = true;
341 
342         var dirty = function() {
343             this.glProgram = null;
344             isDirty = true;
345         }
346         this.dirty = dirty;
347 
348         var clean = function() {
349             isDirty = false;
350         }
351 
352         this.isDirty = function() {
353             return isDirty;
354         }
355         
356         /** Attached shaders */
357         this.shaders   = [];
358 
359         /** Attributes from all attached shaders. */
360         this.attributes = [];
361 
362         /** Uniforms from all attached shaders. */
363         this.uniforms   = [];
364 
365         this._setupAttributes = function() {
366             this.attributes = [];
367 
368             // Each shader keeps track of its attributes,
369             for (var i = 0; i < this.shaders.length; i++) {
370                 var shader = $W.shaders[this.shaders[i]];
371 
372                 for (var j = 0; j < shader.attributes.length; j++) {
373                     var attribute = shader.attributes[j].clone();
374 
375                     // but attribute locations are unique to each program
376                     attribute.location = $W.GL.getAttribLocation(this.glProgram, attribute.name);
377 
378                     // buffers are unique to each object
379                     //attribute.buffer   = $W.GL.createBuffer();
380 
381                     this.attributes.push(attribute);
382                 }
383             }
384         }
385 
386         this._setupUniforms = function() {
387 
388 
389             // Each shader keeps track of its uniforms
390             for (var i = 0; i < this.shaders.length; i++) {
391                 var shader = $W.shaders[this.shaders[i]];
392 
393                 for (var j = 0; j < shader.uniforms.length; j++) {
394                     var uniform = shader.uniforms[j].clone();
395 
396                     // locations are unique to each shader program (I think)
397                     uniform.location = $W.GL.getUniformLocation(this.glProgram, uniform.name);
398                     $W.debug(uniform.name + " " + uniform.location);
399                     this.uniforms.push(uniform);
400                 }
401             }
402         }

404         /** Set the action to take per fram for a particular uniform.
405          * Here instead of the shader because different programs can both
406          * use the same shader, but treat it differently.
407          * XXX What if two shaders have the same named variable
408          * XXX Move to Object? If different objects could need to 
409          * different actions then we ought to store the uniform actions
410          * on a per-object basis.
411          * @param {String} name The name of the uniform.
412          * @param {Function} action The function to call per frame.
413          */
414         this.setUniformAction = function(name, action) {
415             this.use();
416 
417             var uniform;
418             for (var i = 0; i < this.uniforms.length; i++) {
419                 if (this.uniforms[i].name === name) {
420                     uniform = this.uniforms[i];
421                     break;
422                 }
423             }
424 
425             if (uniform === undefined) {
426                 console.error("Cannot set uniform `" + name + "` in shader program `" + this.name + "`, no uniform with that name exists");
427                 return;
428             }else {
429                 $W.info("Setting action for uniform `" + name + "` in shader program `" + this.name + "`");
430             }
431 
432             uniform.action = action;
433         }

435         /** Set the named uniform variable to a single value.
436          * Overloaded to allow 1-4 values to be passed. This allows 
437          * this function to set the value of int, float, vec2, vec3, and
438          * vec4 uniform variables in a shader.
439          * @param {String} name The name of the uniform to set.
440          * @param optional1 A value
441          * @param optional2 A value
442          * @param optional3 A value
443          * @param optional4 A value
444          */
445         this.setUniform = function(name) {
446             // Program must be active before we can do anything with it
447             this.use();
448 
449             var uniform;
450             for (var i = 0; i < this.uniforms.length; i++) {
451                 if (this.uniforms[i].name === name) {
452                     uniform = this.uniforms[i];
453                 }
454             }
455 
456             if (uniform === undefined) {
457                 console.error("Cannot set uniform `" + name + "` in shader program `" + this.name + "`, no uniform with that name exists");
458                 return;
459             }else {
460                 $W.log("Setting uniform `" + name + "` in shader program `" + this.name + "`", $W.LL.DEBUG);
461             }
462 
463             //XXX deal with other types too
464             if (arguments.length === 2) {
465                 var val = arguments[1];
466                 if (uniform.type === "int") {
467                     uniform.action = function() {
468                         $W.GL.uniform1i(this.location, val);
469                     }
470                 }else {
471                     uniform.action = function() {
472                         $W.GL.uniform1f(this.location, val);
473                     }
474                 }
475 
476             } else if (arguments.length === 3) {
477                 uniform.action = function() {
478                     $W.GL.uniform2f(this.location, 
479                         arguments[1], arguments[2]);
480                 }
481 
482             } else if (arguments.length === 4) {
483                 uniform.action = function() {
484                     $W.GL.uniform3f(this.location, 
485                         arguments[1], arguments[2], 
486                         arguments[3]);
487                 }
488 
489             } else if (arguments.length === 5) {
490                 uniform.action = function() {
491                     $W.GL.uniform4f(this.location, 
492                         arguments[1], arguments[2], 
493                         arguments[3], arguments[4]);
494                 }
495 
496             }
497         }

499         /** Called once per frame to calculate and set uniforms. */
500         this.processUniforms = function(obj) {
501             for (var i = 0; i < this.uniforms.length; i++) {
502                 this.uniforms[i].action(this.uniforms[i], obj);
503             }
504         }

506         /** Link this shader program to make it useable.
507          * Will [re]compile all attached shaders if necessary.
508          */
509         this.link = function() {
510             $W.log("linking program `" + this.name + "`");
511 
512 
513             // Only delete the program if one already exists
514             if (this.glProgram !== null) {
515                 $W.log("already exists, deleting and relinking");
516                 $W.GL.deleteProgram(this.glProgram);
517                 this.attributes = [];
518                 this.uniforms = [];
519                 this.glProgram = null;
520             }
521 
522             this.glProgram = $W.GL.createProgram();
523 
524             // Attach all the shaders
525             for (var i = 0; i < this.shaders.length; i++) {
526                 var shader = $W.shaders[this.shaders[i]];
527                 var glShader = shader.getGLShader();
528 
529                 // if the shader is still dirty after calling get,
530                 // which should have cleaned it, then the compile failed.
531                 if (shader.isDirty()) {
532                     $W.warn(this.shaders[i] + " failed to compile");
533                 }else {
534                     $W.GL.attachShader(this.glProgram, glShader);
535                 }	
536             }
537 
538             $W.GL.linkProgram(this.glProgram);
539 
540             // Check for errors
541             if (!$W.GL.getProgramParameter(this.glProgram, $W.GL.LINK_STATUS)) {
542                 console.error("Link error in `" + this.name + "`\n" +
543                         $W.GL.getProgramInfoLog(this.glProgram));
544                 dirty();
545             }
546                 
547             clean();
548             this._setupAttributes();
549             this._setupUniforms();
550 
551             console.groupEnd();
552             return isDirty;
553         }

561         /** Remove the named shader from this program.
562          * Can be used to alter a shader program on the fly.
563          * @param {String} name
564          */
565         this.detachShader = function(name) {
566             var tempShaders = [];
567 
568             for (var i = 0; i < this.shaders.length; i++) {
569                 if (this.shaders[i] != name) {
570                     tempShaders.push(this.shaders[i]);
571                 }
572             }
573 
574             isDirty = true;
575             this.shaders = tempShaders;
576         }

578         /** Attach a shader to this shader program.
579          * Overloaded to work with as <br>
580          * attachShader(GLSL.Shader shader)<br>
581          * attachShader(String name, String source, type)<br>
582          * attachShader(String name, String filePath)<br>
583          * attachShader(String name, String source, type)<br>
584          */
585         this.attachShader = function(shader, path, type) {
586             // Shader from node ID, filename, or source.
587             // Otherwise we've been passed a shader object.
588             if (typeof shader === 'string') { 
589                 // Shader from ID
590                 if (arguments.length === 1) {
591                     shader = new $W.GLSL.Shader(shader);
592 
593                 // Shader from file
594                 }else if (arguments.length === 2){
595 
596                     // Try to infer type
597                     var ext = path.slice(path.length - 4);
598                     if (ext === 'vert' || ext.slice(2) === 'vp') {
599                         type = $W.GL.VERTEX_SHADER;
600                     }
601                     if (ext === 'frag' || ext.slice(2) === 'fp') {
602                         type = $W.GL.FRAGMENT_SHADER;
603                     }
604 
605                     try {
606                         shader = new $W.GLSL.Shader(shader, $W.util.loadFileAsText(path), type); 
607                     }catch (e) {
608                         console.error(e);
609                         return;
610                     }
611 
612                 // Shader from source
613                 }else {
614                     shader = new $W.GLSL.Shader(shader, path, type);
615                 }
616             }
617 
618             this._attachShader(shader);
619         }

621         this._attachShader = function(shader) {
622             $W.log("\tattaching '" + shader.name + "' to '" + this.name + "'", $W.LL.INFO);
623             try{
624                 shader.addProgram(this.name);
625             }catch(e) {
626                 console.error(e);
627             }
628             isDirty = true;
629             this.shaders.push(shader.name);
630         }

644                 if (!this.link()) { return false; }
645             }
646             $W.GL.useProgram(this.glProgram);
647             return true;
