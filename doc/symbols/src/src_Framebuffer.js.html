1 $W.initFramebuffer = function() {
  2 
  3     $W.framebuffers = {};
  4 
  5     $W.Framebuffer = function(name) {
  6         $W.info("Creating framebuffer");
  7         var GL = $W.GL;
  8         var RBUF = GL.RENDERBUFFER;
  9         var FBUF = GL.FRAMEBUFFER;
 10 
 11         this.name = name;
 12         $W.framebuffers[name] = this;
 13         this.glFramebuffer = GL.createFramebuffer();
 14         this.glRenderbuffers = [];
 15         this.textures = [];
 16 
 17         this.isGood = function FBUF_isGood() {
 18             try {
 19                 if (!GL.isFramebuffer(this.glFramebuffer)) {
 20                     throw("Invalid framebuffer");
 21                 }
 22                 var status = GL.checkFramebufferStatus(this.glFramebuffer);
 23                 switch (status) {
 24                     case GL.FRAMEBUFFER_COMPLETE:
 25                         break;
 26                     default:
 27                         throw("Incomplete framebuffer: " + status);
 28                 }
 29             }catch (e) {
 30                 console.error(e);
 31                 return false;
 32             }
 33             return true;
 34         };
 35 
 36         this.bind = function FBUF_bind(){
 37             GL.bindFramebuffer(FBUF, this.glFramebuffer);
 38         };
 39 
 40         this.unbind = function FBUF_unbind(){
 41             GL.bindFramebuffer(FBUF, null);
 42         };
 43 
 44         this.attachRenderbuffer = function FBUF_attachRenderbuffer(storageFormat, width, height, attachment) {
 45             var rBuffer = GL.createRenderbuffer();
 46             this.glRenderbuffers.push(rBuffer);
 47 
 48             this.bind();
 49             GL.bindRenderbuffer(RBUF, rBuffer);
 50             GL.renderbufferStorage(RBUF, storageFormat, width, height);
 51             GL.framebufferRenderbuffer(FBUF, attachment, RBUF, rBuffer);
 52             GL.bindRenderbuffer(RBUF, null);
 53             this.unbind();
 54         };
 55 
 56         this.attachExistingTexture = function FBUF_attachExistingTexture(texture, attachment) {
 57             this.textures.push(texture);
 58             texture.bind();
 59             GL.framebufferTexture2D(GL.FRAMEBUFFER, attachment, GL.TEXTURE_2D, texture.glTexture, 0);
 60             texture.unbind();
 61         }
 62 
 63         this.attachNewTexture = function FBUF_attachNewTexture(format, width, height, attachment) {
 64             var texture = new $W.Texture(this.name + 'Texture' + this.textures.length);
 65 
 66             texture.bind();
 67             try{
 68                 GL.texImage2D(GL.TEXTURE_2D, 0, format, width, height,
 69                         0, format, $W.GL.UNSIGNED_BYTE, null);
 70             } catch (e) {
 71                 console.warn('Using empty texture fallback');
 72                 var storage = new WebGLUnsignedByteArray(4 * width * height);
 73                 GL.texImage2D(GL.TEXTURE_2D, 0, format, width, height,
 74                         0, format, $W.GL.UNSIGNED_BYTE, storage);
 75             }
 76             texture.unbind();
 77 
 78             this.attachExistingTexture(texture, attachment);
 79         }
 80 
 81         this.attachTexture = function FBUF_attachTexture() {
 82             this.bind();
 83             if (arguments.length === 4) {   
 84                 this.attachNewTexture.apply(this, arguments);
 85             }else {
 86                 this.attachExistingTexture.apply(this, arguments);
 87             }
 88             this.unbind();
 89         }
 90     };
 91 };
