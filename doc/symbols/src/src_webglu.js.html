161         /** Load the file at the given path as text.
162          * @param {String} path The path to the file.
163          * @return {String} Data in the file as text.
164          */
165         loadFileAsText:function(path) {
166             $W.info("Loading file `" + path + "`");
167             var xhr = null;
168             xhr = new XMLHttpRequest();
169 
170             if (!xhr) { 
171                 return null; 
172             }
173 
174             xhr.overrideMimeType("text/plain");
175 
176             // Deal with firefox security for file:// urls
177             try {
178                 var nsPM = null;
179                 if (typeof(netscape) !== 'undefined' && 
180                     typeof(netscape.security) !== 'undefined' && 
181                     typeof(netscape.security.PrivilegeManager) !== 'undefined') {
182                     nsPM = netscape.security.PrivilegeManager;
183                 }
184                 if (document.location.href.match(/^file:\/\//)) {
185                     if (nsPM !== null) {
186                         nsPM.enablePrivilege("UniversalBrowserRead");
187                     }
188                 }
189             }catch (e) {
190                 console.error(e);
191                 throw "\tBrowser security may be restrcting access to local files";
192             }
193 
194             try {
195 
196                 xhr.open("GET", path, false);
197 
198                 // Ignore cache if the file is served from localhost or is at a
199                 // file:// url as it's safe to assume that's a developer.
200                 var url = window.location.href;
201                 if (    url.match(/^http:\/\/localhost/) !== null ||
202                         url.match(/^http:\/\/127\.0\.0\.1/) !== null ||
203                         url.match(/^file:/) !== null) {
204 
205                     xhr.setRequestHeader('Pragma', 'Cache-Control: no-cache');
206                 }
207 
208                 xhr.send(null);
209 
210             }catch (e) { 
211                 throw e; 
212             }
213 
214 
215             if (xhr.status !== 200) {
216                 console.error("\tCompleted with status: " + xhr.status);
217                 return "File load error: " + xhr.status;
218             }else {
219                 $W.debug("\tCompleted with status: " + xhr.status);
220                 return xhr.responseText;
221             }
222 
223         },
224 
225         include: function(path) {
226             $W.info('Including ' + path);
227             var script = $W.util.loadFileAsText(path);
228             $W.log(path);
229             window.eval(script);
230         }

234     /** 
235      * Initialize the WebGLU system, optionally initizlizing the WebGL
236      * subsystem.
237      * canvas - the DOM node or id string of a canvas element to init WebGL,
238      *          defaults to using the DOM element with ID 'canvas'.
239      *          Pass `false` to skip init of WebGL subsystem.
240      */
241     initialize:function $W_initialize(canvas) {
242         $W.initLogging();
243         $W.log("Initializing WebGLU");
244 
245         var initModule = function(module) {
246             // If this is not the single file version, load each module
247             if (typeof($W['init' + module]) === 'undefined') {
248                 $W.util.include($W.paths.libsrc + module + '.js');
249             }
250 
251             // In either case call the appropriate init function
252             $W['init' + module]();
253         };
254 
255         // Init Util module early so we can bail out fast.
256         initModule('Util');
257         $W.util.setCanvas(canvas);
258         if (!$W.util.hasWebGLTypes()) {
259            $W.util.displayWebGLFailure(); 
260            return;
261         }
262 
263         var modules = ['Constants', 'DefaultUniformActions', 'GLSL',
264             'GLU', 'Animation', 'Object', 'Texture', 'Framebuffer', 'Material',
265             'Renderer' ];
266 
267         for (var i = 0; i < modules.length; i++) {
268             initModule(modules[i]);
269         }
270 
271         // create the matrix stacks we'll be using to store transformations
272         $W.modelview  = new $W.GLU.MatrixStack();
273         $W.projection = new $W.GLU.MatrixStack();
274 
275         $W.camera     = new $W.Camera();
276         $W.timer      = new $W.Timer();
277 
278         if (canvas === false) {
279             console.log("WebGL init skipped");
280         }else {
281             if ($W.util.initWebGL()) {
282                 $W.camera.aspectRatio = $W.canvas.width / $W.canvas.height;
283                 $W.util.loadDefaultAssets();
284                 return true;
285             }else {
286                 return false;
287             }
288         }
289     },

323 $W.LL = {
324 DEBUG :0,
325        INFO  :1,
326        NOTE  :2,
327        WARN  :3
328 };
329 $W.loglevel = $W.LL.NOTE;
330 $W.initLogging = function() {
331         if (window.console === undefined) {
332             console = {};
333         } // Dummy object
334 
335         if (console.log === undefined) {
336             console.log = function(){};
337         }else {
338             $W.logPrefix = '\t';
339             $W.logIndentAmount = 0;
340             $W.getLogPrefix = function() {
341                 var prefix = '';
342                 for (var i = 0; i < $W.logIndentAmount; i++) {
343                     prefix += $W.logPrefix;
344                 }
345                 return prefix;
346             };
347             $W.indentLog = function(){$W.logIndentAmount++;};
348             $W.dedentLog = function(){$W.logIndentAmount--;};
349                     
350             $W.log = function(message, level) {
351                 if (typeof(level) === 'undefined') {level = $W.LL.NOTE;}
352 
353                 if (level >= $W.loglevel) {
354                     if (typeof(message) === 'string') {
355                         console.log($W.getLogPrefix() + message);
356                     }else {
357                         console.dir(message);
358                     }
359                 }
360             };
361             $W.info = function(message) {
362                 $W.log(message, $W.LL.INFO);
363             };
364             $W.debug = function(message) {
365                 $W.log(message, $W.LL.DEBUG);
366             };
367             $W.warn = function(message) {  
368                 if ($W.LL.WARN >= $W.loglevel) {
369                     console.warn($W.getLogPrefix() + message);
370                 }
371             };
372         }
373 
374         // If console.log exists, but one or more of the others do not,
375         // use console.log in those cases.
376         if (console.dir === undefined) {
377             console.dir           = console.log;
378         }
379         if (console.warn === undefined) {
380             console.warn           = console.log;
381         }
382         if (console.error === undefined) {
383             console.error          = console.log;
384         }
385         if (console.group === undefined) {
386             console.group          = console.log;
387         }
388         if (console.groupCollapsed === undefined) {
389             console.groupCollapsed = console.group;
390         }
391         if (console.groupEnd === undefined) {
392             console.groupEnd       = console.log;
393         }
394     };
395 $W.initLogging();
