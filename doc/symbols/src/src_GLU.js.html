 19         /** glLookAt
 20          * (c) 2009 Vladimir Vukicevic
 21          */
 22         lookAt : function (ex, ey, ez,
 23                            tx, ty, tz,
 24                            ux, uy, uz) {
 25             var eye = $V([ex, ey, ez]);
 26             var target = $V([tx, ty, tz]);
 27             var up = $V([ux, uy, uz]);
 28 
 29             var z = eye.subtract(target).toUnitVector();
 30             var x = up.cross(z).toUnitVector();
 31             var y = z.cross(x).toUnitVector();
 32 
 33             var m = $M([[x.e(1), x.e(2), x.e(3), 0],
 34                     [y.e(1), y.e(2), y.e(3), 0],
 35                     [z.e(1), z.e(2), z.e(3), 0],
 36                     [0, 0, 0, 1]]);
 37 
 38             var t = $M([[1, 0, 0, -ex],
 39                     [0, 1, 0, -ey],
 40                     [0, 0, 1, -ez],
 41                     [0, 0, 0, 1]]);
 42             return m.x(t);
 43         },

 45         /** glOrtho
 46          * (c) 2009 Vladimir Vukicevic
 47          */
 48         ortho : 
 49             function (left, right,
 50                     bottom, top,
 51                     znear, zfar)
 52             {
 53                 var tx = -(right+left)/(right-left);
 54                 var ty = -(top+bottom)/(top-bottom);
 55                 var tz = -(zfar+znear)/(zfar-znear);
 56 
 57                 return $M([[2/(right-left), 0, 0, tx],
 58                         [0, 2/(top-bottom), 0, ty],
 59                         [0, 0, -2/(zfar-znear), tz],
 60                         [0, 0, 0, 1]]);
 61             },
 62 
 63         /** glFrustrum
 64          * (c) 2009 Vladimir Vukicevic
 65          */
 66         frustrum : 
 67             function (left, right,
 68                     bottom, top,
 69                     znear, zfar)
 70             {
 71                 var X = 2*znear/(right-left);
 72                 var Y = 2*znear/(top-bottom);
 73                 var A = (right+left)/(right-left);
 74                 var B = (top+bottom)/(top-bottom);
 75                 var C = -(zfar+znear)/(zfar-znear);
 76                 var D = -2*zfar*znear/(zfar-znear);
 77 
 78                 return $M([[X, 0, A, 0],
 79                         [0, Y, B, 0],
 80                         [0, 0, C, D],
 81                         [0, 0, -1, 0]]);
 82             },
 83 
 84         /** glPerpective
 85          * (c) 2009 Vladimir Vukicevic
 86          */
 87         perspective : 
 88             function (fovy, aspect, znear, zfar)
 89             {
 90                 var ymax = znear * Math.tan(fovy * Math.PI / 360.0);
 91                 var ymin = -ymax;
 92                 var xmin = ymin * aspect;
 93                 var xmax = ymax * aspect;
 94      
 95                 return $W.GLU.frustrum(xmin, xmax, ymin, ymax, znear, zfar);
 96             }

103     /* @class Operates similarly to the standard OpenGL built in matricies. 
104     * However * it is not identical. Rather than calling glMatrixMode, 
105     * you specify the matrix you want to modify prior to the call.
106     * e.g. if `myTranslationVector` is the vector to translate by then to 
107     * translate the ModelView matrix you would call
108     * `$W.modelview.translate(myTranslationVector);`
109     */
110     $W.GLU.MatrixStack = function()  {
111         this._matrixStack = [];
112         this.matrix = Matrix.I(4);
113 
114         /** converts the matrix to the format we need when we send it to 
115          * a shader.
116          * @returns {WebGLFloatArray} The matrix as a flattened array.
117          */
118         this.getForUniform = function() {
119            return new WebGLFloatArray(this.matrix.flatten());
120         };
121 
122 
123         /** glPushMatrix
124          * (c) 2009 Vladimir Vukicevic
125          * @param {Matrix} [m=this.matrix] A Sylvester matrix to add on 
126          * top of the stack.
127          */
128         this.pushMatrix = function (m) {
129             if (m) {
130                 this._matrixStack.push(m.dup());
131                 this.matrix = m.dup();
132             } else {
133                 this._matrixStack.push(this.matrix.dup());
134             }
135         };
136         this.push = this.pushMatrix;
137 
138         /** glPopMatrix
139          * (c) 2009 Vladimir Vukicevic
140          */
141         this.popMatrix = function () {
142             if (this._matrixStack.length === 0) {
143                 throw "Invalid popMatrix!";
144             }
145             this.matrix = this._matrixStack.pop();
146             return this.matrix;
147         };
148         this.pop = this.popMatrix;
149 
150         /** glMultMatrix
151          * (c) 2009 Vladimir Vukicevic
152          * @param {Vector} m A Sylvester matrix to multiply by.
153          */
154         this.multMatrix = function (m) {
155             this.matrix = this.matrix.x(m);
156         };
157         this.multiply = this.multMatrix;
158 
159         /** glTranslate
160          * (c) 2009 Vladimir Vukicevic
161          * @param {Vector} v A Sylvester vector to translate by
162          */
163         this.translate = function (v) {
164             var m = Matrix.Translation($V([v[0],v[1],v[2]])).ensure4x4();
165             this.multMatrix(m);
166         };
167 
168         /** glRotate
169          * (c) 2009 Vladimir Vukicevic
170          * @param {Number} ang Angle to rotate by.
171          * @param {Vector} v A Sylvester vector to rotate around.
172          */
173         this.rotate = function (ang, v) {
174             var arad = ang * Math.PI / 180.0;
175             var m = Matrix.Rotation(arad, $V([v[0], v[1], v[2]])).ensure4x4();
176             this.multMatrix(m);
177         };
178 
179         /** glScale
180          * (c) 2009 Vladimir Vukicevic
181          * @param {Vector} v A Sylvester vector to scale by.
182          */
183         this.scale = function (v) {
184             var m = Matrix.Diagonal([v[0], v[1], v[2], 1]);
185             this.multMatrix(m);
186         };
187 
188         /** invert
189          * (c) 2009 Vladimir Vukicevic
190          */
191         this.invert = function () {
192             this.matrix = this.matrix.inv();
193         };
194 
195         /** glLoadIdentity
196          * (c) 2009 Vladimir Vukicevic
197          */
198         this.loadIdentity = function () {
199             this.matrix = Matrix.I(4);
200         };
201     };
