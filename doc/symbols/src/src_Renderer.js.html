1 $W.initRenderer = function() {
  2     $W.renderer = {};
  3     $W.renderer.bindObjectAttributes = function RNDR_bindObjectAttributes(obj, mat) {
  4         var program = mat.program;
  5         for (var i = 0; i < mat.attributes.length; i++) {
  6             var attrib = mat.attributes[i];
  7             var arrayBuffer = obj.arrayBuffers[attrib.name];
  8 
  9             if (arrayBuffer === null || typeof(arrayBuffer) === 'undefined') {
 10                 console.warn("No data for attribute `" + attrib.name + "`");
 11             } else {
 12                 try {
 13                     arrayBuffer.bind();
 14                     arrayBuffer.associate(attrib);
 15                 }catch (e) {
 16                     console.error(e);
 17                     console.trace();
 18                 }
 19             }       
 20         }
 21     };
 22     $W.renderer.processUniforms = function RNDR_processUniforms(obj, mat) {
 23         for (var i = 0; i < mat.uniforms.length; i++) {
 24             mat.uniforms[i].action(mat.uniforms[i], obj, mat);
 25         }
 26     };
 27     $W.renderer.renderObject = function RNDR_renderObject(obj, mat, drawFun) {
 28         mat.program.use();
 29         this.processUniforms(obj, mat);
 30         this.bindObjectAttributes(obj, mat);
 31         try {
 32             drawFun(obj, mat);
 33         }catch (e) {
 34             console.error("draw error");
 35             console.error(e);
 36         }
 37     };
 38 
 39     $W.renderer.drawArrays = function OBJ_drawArrays(obj, mat) {
 40         try {
 41             $W.GL.drawArrays(obj.type, 0, obj.vertexCount);
 42         }catch (e) {
 43             console.error("drawArrays Failure");
 44             console.error(e);
 45         }
 46     };
 47 
 48     $W.renderer.drawElements = function OBJ_drawElements(obj, mat) {
 49         var elements = obj.arrayBuffers.wglu_internal_elements;
 50         $W.GL.bindBuffer($W.GL.ELEMENT_ARRAY_BUFFER, elements.glBuffer);
 51         try {
 52             $W.GL.drawElements(obj.type, elements.data.length, 
 53                 $W.GL.UNSIGNED_SHORT, elements.data);
 54         }catch (e) {
 55             console.error("drawElements Failure");
 56             console.error(e);
 57         }
 58     };
 59 };
