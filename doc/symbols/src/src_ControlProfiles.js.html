1 /** @ignore Wrapper function to allow multifile or single file organization */
  2 $W.initControlProfiles = function() {
  3     if (typeof($G) === 'undefined') {
  4         $W.useGameGLU();
  5     }
  6 
  7     
  8     /** For dropin control systems
  9      * @namespace Control Profiles
 10      */
 11     $G.profiles = {};
 12         
 13     $W.camera.profileUpdates = [];
 14     $W.camera.update = function() {
 15         for (var i = 0; i < this.profileUpdates.length; i++) {
 16             this.profileUpdates[i](this);
 17         }
 18     };
 19 
 20      
 21     /** @class A input handling and camera updating control profile
 22      * @memberOf $G.profiles
 23      */
 24     $G.profiles.Profile = function() {
 25         this.apply = function Profile_apply() {
 26             this.controls();
 27             this.camera();
 28         };
 29     };
 30 
 31     /** Click-dragging will rotate the camera
 32      */
 33     $G.profiles.DragToRotateCamera = new function() {
 34         $G.profiles.Profile.call(this);
 35 
 36         this.controls = function DragToRotateCamera_controls() {
 37             var st = $G.state;
 38 
 39             st.mouseDown    = false;
 40             st.wasMouseDown = false;
 41             st.lastM  = $V([0,0]);
 42             st.posM   = $V([0,0]);
 43             st.deltaM = $V([0,0]);
 44             st.totalM = $V([0,0]);
 45 
 46             $G.bind('-m1', function(x,y) {
 47                     st.mouseDown = false;
 48             });
 49 
 50             $G.bind('+m1', function(x,y) {
 51                     st.mouseDown = true;
 52                     st.lastM.elements = [x, y];
 53                     st.posM.elements = [x,y];
 54             });
 55 
 56             $G.bind('mousemove', function(x,y) {
 57                 if (st.mouseDown && st.wasMouseDown) {
 58                     // avoid moving around when we click then release, move
 59                     // the mouse, then click again.
 60                     st.lastM.elements = st.posM.elements;
 61                     st.posM.elements = [x, y];
 62                     st.deltaM = st.posM.subtract(st.lastM);
 63                     st.totalM = st.totalM.add(st.deltaM);
 64                 }
 65                 st.wasMouseDown = st.mouseDown;
 66             });
 67 
 68         };
 69 
 70         this.camera = function DragToRotateCamera_camera() {
 71             var st = $G.state;
 72             $W.camera.profileUpdates.push(function(camera) {
 73 
 74                 var x = -1 * st.totalM.e(1);
 75                 var y = st.totalM.e(2);
 76 
 77                 x /= 100;
 78                 y /= 100;
 79 
 80                 camera.target.elements = [Math.cos(x), y, Math.sin(x)];
 81                 st.deltaM.elements = [0,0];
 82             });
 83         };
 84     }();
 85 
 86     /** Scrolling the mouse will zoom the camera
 87      */
 88     $G.profiles.ScrollToZoomCamera = new function() {
 89         $G.profiles.Profile.call(this);
 90 
 91         this.controls = function DragToRotateCamera_controls() {
 92             var st = $G.state;
 93 
 94             st.zoom = $W.camera.yfov;
 95 
 96             $G.bind('mousewheel', function(delta) {
 97                     st.zoom -= delta * 1.5;
 98                     st.zoom = Math.max(10, st.zoom);
 99                     st.zoom = Math.min(90, st.zoom);
100             });
101 
102         };
103 
104         this.camera = function DragToRotateCamera_camera() {
105             var st = $G.state;
106             $W.camera.profileUpdates.push(function(camera) {
107                 camera.yfov = st.zoom;
108             });
109         };
110     }();
111 
112 
113 };
