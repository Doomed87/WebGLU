1 $W.initTexture = function() {
  2     $W.Texture = function(name) {
  3         this.glTexture = $W.GL.createTexture();
  4         this.name = name;
  5         $W.textures[name] = this;
  6         $W.textures.push(this);
  7 
  8         this.bind = function() {
  9             $W.GL.bindTexture($W.GL.TEXTURE_2D, this.glTexture);
 10         };
 11             
 12         this.unbind = function() {
 13             $W.GL.bindTexture($W.GL.TEXTURE_2D, null);
 14         };
 15 
 16         this.update = function(){};
 17 
 18         this.bind();
 19         $W.GL.texParameteri($W.GL.TEXTURE_2D, $W.GL.TEXTURE_MIN_FILTER, $W.GL.LINEAR);
 20         $W.GL.texParameteri($W.GL.TEXTURE_2D, $W.GL.TEXTURE_WRAP_S, $W.GL.CLAMP_TO_EDGE);
 21         $W.GL.texParameteri($W.GL.TEXTURE_2D, $W.GL.TEXTURE_WRAP_T, $W.GL.CLAMP_TO_EDGE);
 22 
 23     };
 24 
 25     $W.CanvasTexture = function(name, src) {
 26         $W.Texture.call(this, name);
 27 
 28         this.canvas = src;
 29         this.canvas.texture = this;
 30 
 31         this.update = function() {
 32             var gl = $W.GL;
 33             this.bind();
 34             gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, gl.RGBA, 
 35                 gl.UNSIGNED_BYTE, this.canvas);
 36             gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.NEAREST);
 37             gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.NEAREST);
 38             this.unbind();
 39         };
 40 
 41     };
 42 
 43     /** A dynamic texture from a `video` element.    
 44      * @param {String} name The global name this texture will be referenced
 45      * by elsewhere.
 46      * @param {String|Video} src Video path or DOM video element.
 47      */
 48     $W.VideoTexture = function(name, src) {
 49         $W.Texture.call(this, name);
 50 
 51         this.setSource = function(video) {
 52             // Path to video
 53             if (typeof(video) === 'string') {
 54                 if (typeof(this.video) === 'undefined') {
 55                     this.video = document.createElement('video');
 56                     document.getElementsByTagName('body')[0].appendChild(this.video);
 57                 }
 58 
 59                 this.video.src = video;
 60 
 61             // DOM Video element
 62             }else {
 63                 this.video = video;
 64             }
 65 
 66             this.video.texture = this;
 67         };
 68 
 69         this.update = function() {
 70             var gl = $W.GL;
 71             this.bind();
 72             try {
 73                 gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, gl.RGBA, 
 74                     gl.UNSIGNED_BYTE, this.video);
 75             }catch(e) {
 76                 // Ignore errors
 77             }
 78             //gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.NEAREST);
 79             //gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.NEAREST);
 80             //gl.generateMipmap(gl.TEXTURE_2D);
 81             //gl.bindTexture(gl.TEXTURE_2D, null); // clean up after ourselves
 82         }
 83 
 84         this.setSource(src);
 85     };
 86 
 87     /** A static texture from an image file.
 88      * @param {String} name The global name this texture will be referenced
 89      * by elsewhere.
 90      * @param {String} src Path to image file.
 91      */
 92     $W.ImageTexture = function(name, src) {
 93         $W.Texture.call(this, name);
 94         this.image = document.createElement('img');
 95         this.image.texture = this;
 96 
 97         this.image.onload = function() {
 98             var gl = $W.GL;
 99             $W.debug('Loaded texture `' + name + "`");
100             this.texture.bind();
101 
102             gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, gl.RGBA, 
103                 gl.UNSIGNED_BYTE, this.texture.image);
104             gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.NEAREST);
105             gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.NEAREST);
106 
107             this.texture.unbind();
108         }
109 
110         this.setSource = function(src) {
111             this.image.src = src;
112         };
113 
114         if (src !== undefined) {
115             this.setSource(src);
116         }
117     };
118 };
