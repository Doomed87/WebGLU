 13         };
 14 
 15     ProjectionAction = 
 16         function DUA_ProjectionAction(_, object, material) {

 18                     $W.projection.getForUniform());

 22         function DUA_NormalMatrixAction(_, object, material){

 27     SimpleSamplerAction = 
 28         function DUA_SimpleSamplerAction(_, object, material) {
 29             try {
 30                 var gl = $W.GL;
 31                 gl.activeTexture(gl.TEXTURE0);
 32                 $W.textures[object.textures[0]].bind();
 33                 gl.uniform1i(this.location, 0);
 34             }catch (e) {
 35                 console.error("Simple texture uniform error");
 36                 console.error(e);
 37             }
 38         };
 39 
 40     MaterialTextureAction = 
 41         function DUA_MaterialTextureAction(_, object, material) {
 42             try {
 43                 $W.GL.activeTexture($W.GL.TEXTURE0);
 44                 if (typeof(material.textures[0]) === 'undefined') {
 45                     $W.textures.wglu_internal_missing_texture.bind();
 46                 }else {
 47                     $W.textures[material.textures[0]].bind();
 48                 }
 49                 $W.GL.uniform1i(this.location, 0);
 50             }catch (e) {
 51                 console.error("Material texture uniform error");
 52                 console.error(e);
 53             }
 54         };
 55 
 56     genMultiTextureAction = function DUA_genMultiTextureAction(texNum) {

 60 "               $W.GL.activeTexture($W.GL.TEXTURE0);\n"+
 61 "               if (typeof(material.textures["+texNum+"]) === 'undefined') {\n"+
 62 "                   $W.textures.wglu_internal_missing_texture.bind();\n"+
 63 "               }else {\n"+
 64 "                   $W.textures[material.textures["+texNum+"]].bind();\n"+
 65 "               }\n"+

 67 "           }catch (e) {\n"+
 68 "               console.error('Material texture uniform error');\n"+

 71 "        }");

 73     };
 74

 76

 88

 92         }
 93 
 94         return action;
 95     };
