  6      * @param {String} path The path to the file.
  7      * @return {String} Data in the file as text.

  9     $W.util.loadFileAsText = function(path) {
 10         $W.info("Loading file `" + path + "`");
 11         var xhr = null;
 12         xhr = new XMLHttpRequest();
 13 
 14         if (!xhr) { 
 15             return null; 
 16         }
 17 
 18         xhr.overrideMimeType("text/plain");
 19 
 20         // Deal with firefox security for file:// urls
 21         try {
 22             var nsPM = null;
 23             if (typeof(netscape) !== 'undefined' && 
 24                 typeof(netscape.security) !== 'undefined' && 
 25                 typeof(netscape.security.PrivilegeManager) !== 'undefined') {
 26                 nsPM = netscape.security.PrivilegeManager;
 27             }
 28             if (document.location.href.match(/^file:\/\//)) {
 29                 if (nsPM !== null) {
 30                     nsPM.enablePrivilege("UniversalBrowserRead");
 31                 }
 32             }
 33         }catch (e) {
 34             console.error(e);
 35             throw "\tBrowser security may be restrcting access to local files";
 36         }
 37 
 38         try {
 39 
 40             xhr.open("GET", path, false);
 41 
 42             // Ignore cache if the file is served from localhost or is at a
 43             // file:// url as it's safe to assume that's a developer.
 44             var url = window.location.href;
 45             if (    url.match(/^http:\/\/localhost/) !== null ||
 46                     url.match(/^http:\/\/127\.0\.0\.1/) !== null ||
 47                     url.match(/^file:/) !== null) {
 48 
 49                 xhr.setRequestHeader('Pragma', 'Cache-Control: no-cache');
 50             }
 51 
 52             xhr.send(null);
 53 
 54         }catch (e) { 
 55             throw e; 
 56         }
 57 
 58         $W.debug("\tCompleted with status: " + xhr.status);
 59 
 60         return xhr.responseText;
 61 
 62     };

228         var c2 = Math.cos(rotation.e(1) / 2); // c2 = cos(attitude / 2)

240         result.axis[1] = s1*c2*c3 + c1*s3*s3;
241         result.axis[2] = c1*s2*c3 - s1*c2*s3;
242 
243         // Normalize
244         var mag = Math.sqrt(result.axis[0]*result.axis[0] + result.axis[1]*result.axis[1] + result.axis[2]*result.axis[2]);
245         //if (Math.abs(result.axis[0]) > 1) result.axis[0] /= mag;

252     /** Generate vertices, normals, texture coordinates, and element 
253      * indices for a sphere. Intended to be rendered with setElements.
254      * XXX Not a 'sliced' sphere, need to fix that.
255      * @param rings Number of horizontal rings that make up the sphere.
256      * @param slices Number of triangles per ring.
257      * @param [r] The radius of the sphere, defaults to 1.0 if omitted.
258      *
259      * @returns sphere Object containing sphere data.
260      * @returns {Array} sphere.vertices Vertices for sphere of radius r.
261      * @returns {Array} sphere.normals Normals for sphere with given number
262      * of rings and slices.
263      * @returns {Array} sphere.texCoords Texture coordinates for sphere 
264      * with given number of rings and slices.
265      * @returns {Array} sphere.indices Per element indices.
266      */
267     $W.util.genSphere= function UTIL_genSphere(rings, slices, r) {
268         // Default to 10 rings and 10 slices
269         if (rings === undefined) { var rings = 10; }
270         if (slices === undefined) { var slices = 10; }
271         // Default to unit sphere
272         if (r === undefined) { var r = 1; }
273 
274         var sphere = {};
275         sphere.vertices = [];
276         sphere.normals  = [];
277         sphere.texCoords= [];
278         sphere.indices  = [];
279 
280         var piOverRings = Math.PI / rings;
281         var piOverSlices = Math.PI / slices;
282         var genTri = function(ring, slice) {
283                 var theta    = ring * piOverRings;
284                 var phi      = slice * 2 * piOverSlices
285 
286                 var sinTheta = Math.sin(theta);
287                 var cosTheta = Math.cos(theta);
288 
289                 var sinPhi   = Math.sin(phi);
290                 var cosPhi   = Math.cos(phi);
291 
292                 var x = cosPhi * sinTheta;
293                 var y = cosTheta;
294                 var z = sinPhi * sinTheta;
295 
296                 var u = slice / slices;
297                 var v = ring / rings;
298                 
299                 sphere.vertices = sphere.vertices.concat([r*x, r*y, r*z]);
300                 sphere.normals = sphere.normals.concat([x, y, z]);
301                 sphere.texCoords = sphere.texCoords.concat([u,v]);
302         };
303 
304         var genIndices = function(ring, slice) {
305                 var a = (ring * slices) + ring + slice;
306                 var b = a + 1;
307                 var c = b + slices;
308                 var d = c + 1;
309                 sphere.indices = sphere.indices.concat([a,b,c, c,d,b]);
310         };
311 
312         for (var ring = 0; ring <= rings; ring++) {
313             for (var slice = 0; slice <= slices; slice++) {
314                 genTri(ring, slice);
315                 if (ring != rings && slice != slices) {
316                     genIndices(ring, slice);
317                 }
318             }
319         }
320 
321         return sphere;
322     };

324     /** Spherical linear interpolation. For interpolating quaternions.
325      * Based on reference implementation at http://www.euclideanspace.com/maths/algebra/realNormedAlgebra/quaternions/slerp/index.htm
326      * @param {Number} t How far along to interpolate.
327      * @param {Quaternion} q1 {@link Quaternion} to interpolate from.
328      * @param {Quaternion} q2 {@link Quaternion} to interpolate to.
329      */
330     $W.util.slerp=function UTIL_slerp(t, q1, q2) {
331         var result = new $W.Quaternion();
332 
333         var cosHalfTheta = q1.w * q2.w + q1.x * q2.x + q1.y * q2.y + q1.z * q2.z;
334 
335         // q1 == q2
336         if (Math.abs(cosHalfTheta) >= 1) {
337             result.w = q1.w;
338             result.x = q1.x;
339             result.y = q1.y;
340             result.z = q1.z;
341 
342         }else {
343             var halfTheta = Math.acos(cosHalfTheta);
344             var sinHalfTheta = Math.sqrt(1 - cosHalfTheta * cosHalfTheta);
345 
346             // Theta = 180, any direction is valid
347             if (Math.abs(sinHalfTheta) < 0.001) {
348                 result.w = q1.w * 0.5 + q2.w * 0.5;
349                 result.x = q1.x * 0.5 + q2.x * 0.5;
350                 result.y = q1.y * 0.5 + q2.y * 0.5;
351                 result.z = q1.z * 0.5 + q2.z * 0.5;
352 
353             // Spherical linear interpolation
354             }else {
355                 // Ratios
356                 var ratio1 = Math.sin((1 - t) * halfTheta) / sinHalfTheta;
357                 var ratio2 = Math.sin(t * halfTheta) / sinHalfTheta;
358 
359                 result.w = q1.w * ratio1 + q2.w * ratio2;
360                 result.x = q1.x * ratio1 + q2.x * ratio2;
361                 result.y = q1.y * ratio1 + q2.y * ratio2;
362                 result.z = q1.z * ratio1 + q2.z * ratio2;
363             }
364         }
365 
366         return result;
367     };

457     /* Calculate normals at each vertex in vertices, by looking
458      * at triangles formed by every face and averaging.
459      * (c) 2009 Vladimir Vukicevic
460      */
461     $W.util.calculateNormals = function(vertices, faces) {
462         var nvecs;
463 
464         if (vertices[0].length == 3) {
465             nvecs = new Array(vertices.length);
466 
467             for (var i = 0; i < faces.length; i++) {
468                 var j0 = faces[i][0];
469                 var j1 = faces[i][1];
470                 var j2 = faces[i][2];
471 
472                 var v1 = $V(vertices[j0]);
473                 var v2 = $V(vertices[j1]);
474                 var v3 = $V(vertices[j2]);
475 
476 
477                 var va = v2.subtract(v1);
478                 var vb = v3.subtract(v1);
479 
480                 var n = va.cross(vb).toUnitVector();
481 
482                 if (!nvecs[j0]) nvecs[j0] = [];
483                 if (!nvecs[j1]) nvecs[j1] = [];
484                 if (!nvecs[j2]) nvecs[j2] = [];
485 
486                 nvecs[j0].push(n);
487                 nvecs[j1].push(n);
488                 nvecs[j2].push(n);
489             }
490 
491         }else { // handle flattened arrays
492             nvecs = new Array(vertices.length / 3)
493 
494             for (var i = 0; i < faces.length; i+=3) {
495                 var j0 = faces[i+0];
496                 var j1 = faces[i+1];
497                 var j2 = faces[i+2];
498 
499                 var v1 = $V([vertices[j0], vertices[j0+1], vertices[j0+2]]);
500                 var v2 = $V([vertices[j1], vertices[j2+1], vertices[j2+2]]);
501                 var v3 = $V([vertices[j2], vertices[j2+1], vertices[j2+2]]);
502 
503 
504                 var va = v2.subtract(v1);
505                 var vb = v3.subtract(v1);
506 
507                 var n = va.cross(vb).toUnitVector();
508 
509                 if (!nvecs[j0]) nvecs[j0] = [];
510                 if (!nvecs[j1]) nvecs[j1] = [];
511                 if (!nvecs[j2]) nvecs[j2] = [];
512 
513                 nvecs[j0].push(n);
514                 nvecs[j1].push(n);
515                 nvecs[j2].push(n);
516             }
517         }
518 
519         var normals = new Array(vertices.length);
520 
521         // now go through and average out everything
522         for (var i = 0; i < nvecs.length - 1; i++) {
523             var count = nvecs[i].length;
524             var x = 0;
525             var y = 0;
526             var z = 0;
527 
528             for (var j = 0; j < count; j++) {
529                 x += nvecs[i][j].elements[0];
530                 y += nvecs[i][j].elements[1];
531                 z += nvecs[i][j].elements[2];
532             }
533 
534             normals[i] = [x/count, y/count, z/count];
535         }
536 
537         return normals;
538     }

588         Matrix.prototype.ensure4x4 = function MTX_ensure4x4()
589         {
590             if (this.elements.length == 4 && 
591                     this.elements[0].length == 4) {
592                 return this;
593             }
594 
595             if (this.elements.length > 4 ||
596                     this.elements[0].length > 4) {
597                 return null;
598             }
599 
600             for (var i = 0; i < this.elements.length; i++) {
601                 for (var j = this.elements[i].length; j < 4; j++) {
602                     if (i == j) {
603                         this.elements[i].push(1);
604                     }else {
605                         this.elements[i].push(0);
606                     }
607                 }
608             }
609 
610             for (var i = this.elements.length; i < 4; i++) {
611                 if (i === 0) {
612                     this.elements.push([1, 0, 0, 0]);
613                 }else if (i == 1) {
614                     this.elements.push([0, 1, 0, 0]);
615                 }else if (i == 2) {
616                     this.elements.push([0, 0, 1, 0]);
617                 }else if (i == 3) {
618                     this.elements.push([0, 0, 0, 1]);
619                 }
620             }

664     /** @class Quaternion implementation.
665      * based on reference implementation at 
666      * http://3dengine.org/Quaternions
667      * 
668      * Constructor has three overloads:<br><br>
669      * Quaternion(x, y, z, theta); // From axis-angle<br>
670      * Quaternion([w, x, y, z]); // From another quaternion<br>
671      * Quaternion(); // With zeroed rotation
672      */
673     $W.Quaternion = function() {
674         // Called as Quaternion(x, y, z, theta)
675         if (arguments.length == 4) {
676             var angle = (arguments[3] / 180.0) * Math.PI;
677             var result = Math.sin(angle / 2);
678             this.w = Math.cos(angle / 2);
679             this.x = arguments[0] * result;
680             this.y = arguments[1] * result;
681             this.z = arguments[2] * result;
682 
683         // Called as Quaternion([w, x, y, z])
684         }else if (arguments[0] !== undefined && arguments[0].length == 4) {
685             this.w = arguments[0][0];
686             this.x = arguments[0][1];
687             this.y = arguments[0][2];
688             this.z = arguments[0][3];
689 
690         // Called as Quaternion()
691         }else {
692             this.w = 1;
693             this.x = 0;
694             this.y = 0;
695             this.z = 0;
696         }
697 
698         /** @returns {Matrix} The 4x4 rotation matrix equivalent of this
699          * quaternion
700          */
701         this.matrix = function() {
702             var w = this.w;
703             var x = this.x;
704             var y = this.y;
705             var z = this.z;
706             var xx = x * x;
707             var yy = y * y;
708             var zz = z * z;
709             var xy = x * y;
710             var xz = x * z;
711             var xw = x * w;
712             var yz = y * z;
713             var yw = y * w;
714             var zw = z * w;
715 
716             var m = [[],[],[],[0,0,0,1]];
717 
718             m[0][0] = 1 - 2 * (yy + zz);
719             m[0][1] = 2 * (xy + zw);
720             m[0][2] = 2 * (xz - yw);
721             m[0][3] = 0;
722             m[1][0] = 2 * (xy - zw);
723             m[1][1] = 1 - 2 * (xx + zz);
724             m[1][2] = 2 * (yz + xw);
725             m[1][3] = 0;
726             m[2][0] = 2 * (xz + yw);
727             m[2][1] = 2 * (yz - xw);
728             m[2][2] = 1 - 2 * (xx + yy);
729             m[2][3] = 0;
730             
731             return $M(m);
732         };
733 
734         /** Multiply by another quaternion and return the result in a new
735          * quaternion
736          * @param {$W.Quaternion} q The other quaternion
737          */
738         this.multiply = function(q) {
739             var result = new $W.Quaternion();
740             result.w = this.w * q.w - this.x * q.x - this.y * q.y - this.z * q.z;
741             result.x = this.w * q.x + this.x * q.w + this.y * q.z - this.z * q.y;
742             result.y = this.w * q.y + this.y * q.w + this.z * q.x - this.x * q.z;
743             result.z = this.w * q.z + this.z * q.w + this.x * q.y - this.y * q.x;
744             return result;
745         };
746 
747         /* Multiply by another quaternion and store the result in this
748          * quaternion
749          * @param {$W.Quaternion} q The other quaternion
750          */
751         this.multiplied = function(q) {
752             this.w = this.w * q.w - this.x * q.x - this.y * q.y - this.z * q.z;
753             this.x = this.w * q.x + this.x * q.w + this.y * q.z - this.z * q.y;
754             this.y = this.w * q.y + this.y * q.w + this.z * q.x - this.x * q.z;
755             this.z = this.w * q.z + this.z * q.w + this.x * q.y - this.y * q.x;
756         };
757     };

801     /** @class Keeps track of time since application start.
802      * Provides delta time between ticks and tracks FPS.
803      * @property {Number} age The time passed since this timer was started to
804      * the time of the most recent tick in milliseconds.
805      * @property {Number} t The current application time in milliseconds. 
806      * @property {Number} dt The delta time between the previous two ticks. 
807      * @property {Number} pt The time of the previous tick.
808      * @property {Number} frameAvgCount Number of frames to average over.
809      * @property {Number} mspf Milliseconds per frame.
810      * @property {Number} fps Frames per second. 
811      */
812     $W.Timer = function () {
813         this.age = 0;
814         this.t  = (new Date()).getTime();
815         this.dt = 0;
816         this.pt = this.t;
817 
818         /** Update the timer */
819         this.tick = function() {
820             this.t = (new Date()).getTime();
821             this.dt = this.t - this.pt;
822             this.pt = this.t;
823             this.age += this.dt;
824             this.updateFPSCount(this.dt);
825         }
826 
827         /** The time passed since this timer was started to the time of
828          * the most recent tick in seconds.
829          * @returns {Number} seconds
830          */
831         this.ageInSeconds = function() {
832             return this.age / 1000;
833         }
834 
835         // frame timing statistics
836         this.frameAvgCount = 20;  
837         this.mspf= 0; 
838         this.fps = 0;
839 
840         this.recentFPS = []; // last several FPS calcs to average over
841 
842         /** Update the FPS. */
843         this.updateFPSCount = function(dt) {
844             this.mspf += dt; // add this past frame time and renormalize
845             this.mspf /= 2;
846 
847             if (this.recentFPS.unshift(Math.floor(1000 / this.mspf)) > this.frameAvgCount) {
848                 this.recentFPS.pop();
849             } // average FPS over the past frameAvgCount frames
850             
851             this.fps = 0;
852             for (var i = 0; i < this.recentFPS.length; i++) {
853                 this.fps += this.recentFPS[i];
854             }
855             this.fps /= this.recentFPS.length;
856         }
857     };
